/**
 * Bytewise Forgot Password Screen
 * 
 * Password reset form with brand styling
 * Features:
 * - Email input validation
 * - Brand-consistent styling
 * - Toast notifications
 * - Navigation back to login
 */

import React from 'react';
import { BrandStandard } from '../Brand';

interface ForgotPasswordScreenProps {
  onResetRequest: (email: string) => void;
  onNavigateToLogin: () => void;
  showToast: (message: string) => void;
}

export function ForgotPasswordScreen({ 
  onResetRequest, 
  onNavigateToLogin, 
  showToast 
}: ForgotPasswordScreenProps) {
  const handleSubmit = (e: React.FormEvent<HTMLFormElement>) => {
    e.preventDefault();
    const formData = new FormData(e.target as HTMLFormElement);
    const email = formData.get('email') as string;
    
    if (email) {
      showToast('Sending reset link...');
      onResetRequest(email);
    } else {
      showToast('Please enter a valid email address');
    }
  };

  return (
    <div className="min-h-screen bg-background flex items-center justify-center p-6 animate-fade-in">
      <div className="max-w-md w-full">
        <BrandStandard size="lg" className="mb-12" />
        
        <div className="text-center mb-10">
          <h1 
            className="text-brand-heading" 
            style={{ fontFamily: "'League Spartan', sans-serif", fontSize: "1.5rem", fontWeight: 700 }}
          >
            Reset Password
          </h1>
          <p 
            className="text-muted-foreground text-brand-body" 
            style={{ fontFamily: "'Quicksand', sans-serif", fontSize: "0.875rem", fontWeight: 400 }}
          >
            We'll send you a link to reset your password
          </p>
        </div>
        
        <form className="space-y-6" onSubmit={handleSubmit}>
          <div>
            <label 
              className="block text-foreground mb-3 text-brand-label" 
              style={{ fontFamily: "'Work Sans', sans-serif", fontSize: "0.875rem", fontWeight: 500 }}
            >
              Email address
            </label>
            <input
              type="email"
              name="email"
              placeholder="Enter your email"
              className="w-full p-4 rounded-lg border border-border bg-input-background focus:outline-none focus:ring-2 focus:ring-primary focus:border-primary transition-all text-brand-body"
              style={{ fontFamily: "'Quicksand', sans-serif", fontSize: "1rem", fontWeight: 400 }}
              required
            />
          </div>
          
          <button 
            type="submit"
            className="w-full p-4 bg-primary text-primary-foreground rounded-lg btn-animate hover:bg-primary/90 focus:outline-none focus:ring-2 focus:ring-primary focus:ring-offset-2 text-brand-button"
            style={{ fontFamily: "'Work Sans', sans-serif", fontSize: "1rem", fontWeight: 500 }}
          >
            Send Reset Link
          </button>
          
          <div className="text-center pt-4">
            <button
              type="button"
              onClick={() => {
                console.log('🧭 Returning to login from forgot password');
                onNavigateToLogin();
              }}
              className="text-primary hover:text-primary/80 transition-colors text-brand-button"
              style={{ fontFamily: "'Work Sans', sans-serif", fontSize: "0.875rem", fontWeight: 500 }}
            >
              ← Back to Login
            </button>
          </div>
        </form>
      </div>
    </div>
  );
}
