/**
 * Bytewise Calendar Screen Component
 * 
 * Enhanced calendar with real meal data integration and debug capabilities
 * Features:
 * - Real-time data from meal logging system
 * - Daily nutrition summary with actual logged meals
 * - Visual progress indicators for each day
 * - Interactive date selection with meal details
 * - Brand-consistent design following Bytewise guidelines
 * - Mobile-optimized responsive layouts
 * - FIXED: Now uses real meal data from both localStorage sources
 * - ADDED: Debug capabilities to track data flow issues
 */

import { useState, useEffect } from 'react';
import { Calendar, ChevronLeft, ChevronRight, Plus, TrendingUp, Flame, Utensils, Target, RefreshCw, AlertCircle, Database } from 'lucide-react';
import { Button } from '../ui/button';
import { Badge } from '../ui/badge';
import { ImageWithFallback } from '../figma/ImageWithFallback';

interface CalendarScreenProps {
  onNavigate?: (tab: string) => void;
}

interface DayMealData {
  meals: number;
  calories: number;
  protein: number;
  carbs: number;
  fat: number;
  status: 'complete' | 'partial' | 'empty';
  breakdown: {
    breakfast: number;
    lunch: number;
    dinner: number;
    snacks: number;
  };
}

export function CalendarScreen({ onNavigate }: CalendarScreenProps) {
  const [selectedDate, setSelectedDate] = useState(new Date());
  const [currentMonth, setCurrentMonth] = useState(new Date());
  const [mealData, setMealData] = useState<{ [key: string]: DayMealData }>({});
  const [isLoading, setIsLoading] = useState(true);
  const [debugInfo, setDebugInfo] = useState({
    dailyMealLogsCount: 0,
    bytewiseMealEntriesCount: 0,
    totalDates: 0,
    lastLoadTime: ''
  });

  const daysOfWeek = ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'];
  const monthNames = [
    'January', 'February', 'March', 'April', 'May', 'June',
    'July', 'August', 'September', 'October', 'November', 'December'
  ];

  // Load real meal data from localStorage with enhanced debugging
  useEffect(() => {
    loadMealData();
    
    // Listen for meal data updates
    const handleDataUpdate = () => {
      console.log('📅 Calendar: Meal data updated, reloading...');
      loadMealData();
    };

    window.addEventListener('bytewise-meal-logged', handleDataUpdate);
    window.addEventListener('bytewise-demo-data-loaded', handleDataUpdate);
    window.addEventListener('bytewise-data-updated', handleDataUpdate);
    window.addEventListener('storage', handleDataUpdate);

    return () => {
      window.removeEventListener('bytewise-meal-logged', handleDataUpdate);
      window.removeEventListener('bytewise-demo-data-loaded', handleDataUpdate);
      window.removeEventListener('bytewise-data-updated', handleDataUpdate);
      window.removeEventListener('storage', handleDataUpdate);
    };
  }, []);

  const loadMealData = () => {
    setIsLoading(true);
    
    try {
      console.log('📅 Loading Calendar meal data...');
      
      // Check both possible data sources
      const savedMealLogs = JSON.parse(localStorage.getItem('dailyMealLogs') || '{}');
      const bytewiseMealEntries = JSON.parse(localStorage.getItem('bytewise-meal-entries') || '{}');
      
      console.log('📅 Available meal logs (dailyMealLogs):', Object.keys(savedMealLogs));
      console.log('📅 Available meal entries (bytewise-meal-entries):', Object.keys(bytewiseMealEntries));
      
      // Enhanced debug information
      const debugData = {
        dailyMealLogsCount: Object.keys(savedMealLogs).length,
        bytewiseMealEntriesCount: Object.keys(bytewiseMealEntries).length,
        totalDates: 0,
        lastLoadTime: new Date().toLocaleTimeString()
      };
      
      const processedData: { [key: string]: DayMealData } = {};
      
      // Process both data sources
      const allDates = new Set([
        ...Object.keys(savedMealLogs),
        ...Object.keys(bytewiseMealEntries)
      ]);
      
      debugData.totalDates = allDates.size;
      setDebugInfo(debugData);
      
      console.log('📅 Processing', allDates.size, 'unique dates from both sources');
      
      allDates.forEach(date => {
        let dayLog = savedMealLogs[date];
        
        // If no data in dailyMealLogs, check bytewise-meal-entries and convert
        if (!dayLog?.meals && bytewiseMealEntries[date]) {
          console.log('📅 Converting bytewise data for date:', date);
          const bytewiseEntry = bytewiseMealEntries[date];
          dayLog = {
            date,
            meals: {
              breakfast: bytewiseEntry.breakfast || [],
              lunch: bytewiseEntry.lunch || [],
              dinner: bytewiseEntry.dinner || [],
              snack: bytewiseEntry.snacks || [] // Note: bytewise uses 'snacks', we need 'snack'
            },
            totals: { calories: 0, protein: 0, carbs: 0, fat: 0, fiber: 0, sugar: 0, sodium: 0 }
          };
        }
        
        if (dayLog?.meals) {
          // Calculate totals from meals if not already calculated
          let totals = dayLog.totals;
          if (!totals || totals.calories === 0) {
            const allMeals = [
              ...(dayLog.meals.breakfast || []),
              ...(dayLog.meals.lunch || []),
              ...(dayLog.meals.dinner || []),
              ...(dayLog.meals.snack || [])
            ];
            
            totals = allMeals.reduce((acc, meal) => ({
              calories: acc.calories + (meal.calories || 0),
              protein: acc.protein + (meal.protein || 0),
              carbs: acc.carbs + (meal.carbs || 0),
              fat: acc.fat + (meal.fat || 0),
              fiber: acc.fiber + (meal.fiber || 0),
              sugar: acc.sugar + (meal.sugar || 0),
              sodium: acc.sodium + (meal.sodium || 0)
            }), { calories: 0, protein: 0, carbs: 0, fat: 0, fiber: 0, sugar: 0, sodium: 0 });
          }
          
          const totalMeals = (dayLog.meals.breakfast?.length || 0) +
                            (dayLog.meals.lunch?.length || 0) +
                            (dayLog.meals.dinner?.length || 0) +
                            (dayLog.meals.snack?.length || 0);
          
          // Calculate meal breakdown calories
          const breakfastCalories = (dayLog.meals.breakfast || []).reduce((sum: number, meal: any) => sum + (meal.calories || 0), 0);
          const lunchCalories = (dayLog.meals.lunch || []).reduce((sum: number, meal: any) => sum + (meal.calories || 0), 0);
          const dinnerCalories = (dayLog.meals.dinner || []).reduce((sum: number, meal: any) => sum + (meal.calories || 0), 0);
          const snacksCalories = (dayLog.meals.snack || []).reduce((sum: number, meal: any) => sum + (meal.calories || 0), 0);
          
          // Determine status
          let status: 'complete' | 'partial' | 'empty' = 'empty';
          if (totalMeals >= 3) {
            status = 'complete';
          } else if (totalMeals >= 1) {
            status = 'partial';
          }
          
          processedData[date] = {
            meals: totalMeals,
            calories: Math.round(totals.calories),
            protein: Math.round(totals.protein),
            carbs: Math.round(totals.carbs),
            fat: Math.round(totals.fat),
            status,
            breakdown: {
              breakfast: Math.round(breakfastCalories),
              lunch: Math.round(lunchCalories),
              dinner: Math.round(dinnerCalories),
              snacks: Math.round(snacksCalories)
            }
          };
          
          console.log('📅 Processed day', date, ':', processedData[date]);
        }
      });
      
      console.log('📅 Final processed meal data:', processedData);
      setMealData(processedData);
      
    } catch (error) {
      console.error('❌ Failed to load calendar meal data:', error);
    } finally {
      setIsLoading(false);
    }
  };

  const getDaysInMonth = (date: Date) => {
    const year = date.getFullYear();
    const month = date.getMonth();
    const firstDay = new Date(year, month, 1);
    const lastDay = new Date(year, month + 1, 0);
    const daysInMonth = lastDay.getDate();
    const startingDayOfWeek = firstDay.getDay();

    const days = [];
    
    // Add empty cells for days before the first day of the month
    for (let i = 0; i < startingDayOfWeek; i++) {
      days.push(null);
    }
    
    // Add all days of the month
    for (let day = 1; day <= daysInMonth; day++) {
      days.push(new Date(year, month, day));
    }
    
    return days;
  };

  const formatDateKey = (date: Date) => {
    return date.toISOString().split('T')[0];
  };

  const isToday = (date: Date) => {
    const today = new Date();
    return date.toDateString() === today.toDateString();
  };

  const isSameDate = (date1: Date, date2: Date) => {
    return date1.toDateString() === date2.toDateString();
  };

  const days = getDaysInMonth(currentMonth);

  const navigateMonth = (direction: 'prev' | 'next') => {
    const newMonth = new Date(currentMonth);
    if (direction === 'prev') {
      newMonth.setMonth(newMonth.getMonth() - 1);
    } else {
      newMonth.setMonth(newMonth.getMonth() + 1);
    }
    setCurrentMonth(newMonth);
  };

  const selectedDateData = mealData[formatDateKey(selectedDate)];

  // Calculate weekly stats from real data
  const calculateWeeklyStats = () => {
    const today = new Date();
    const last7Days = [];
    
    for (let i = 6; i >= 0; i--) {
      const date = new Date(today);
      date.setDate(date.getDate() - i);
      last7Days.push(formatDateKey(date));
    }
    
    const weekData = last7Days.map(date => mealData[date]).filter(Boolean);
    
    const totalMeals = weekData.reduce((sum, day) => sum + day.meals, 0);
    const avgCalories = weekData.length > 0 ? Math.round(weekData.reduce((sum, day) => sum + day.calories, 0) / weekData.length) : 0;
    const daysWithMeals = weekData.filter(day => day.meals > 0).length;
    const goalMetPercentage = Math.round((daysWithMeals / 7) * 100);
    
    return {
      totalMeals,
      avgCalories,
      goalMetPercentage
    };
  };

  const weeklyStats = calculateWeeklyStats();

  // Manual refresh function for debugging
  const handleRefreshData = () => {
    console.log('🔄 Manual calendar refresh triggered');
    loadMealData();
  };

  // Debug function to check localStorage
  const debugLocalStorage = () => {
    console.log('🐛 CALENDAR DEBUG INFO:');
    console.log('📅 Selected date:', formatDateKey(selectedDate));
    console.log('🗄️ dailyMealLogs:', localStorage.getItem('dailyMealLogs'));
    console.log('🗄️ bytewise-meal-entries:', localStorage.getItem('bytewise-meal-entries'));
    console.log('📊 Processed meal data:', mealData);
    console.log('📈 Debug info:', debugInfo);
    alert('🐛 Debug info logged to console. Check the browser console for details.');
  };

  return (
    <div className="p-4 pb-24 max-w-md mx-auto animate-fade-in">
      {/* Hero Section with Seamless Header Integration */}
      <div className="relative -mx-4 mb-6">
        <div className="h-64 relative overflow-hidden">
          <ImageWithFallback
            src="https://images.unsplash.com/photo-1490645935967-10de6ba17061?ixlib=rb-4.0.3&auto=format&fit=crop&w=800&h=600&q=80"
            alt="Nutritious meal spread and calendar planning"
            className="absolute inset-0 w-full h-full object-cover"
          />
          <div className="absolute inset-0 bg-gradient-to-t from-black/80 via-black/40 to-transparent"></div>
          
          <div className="absolute inset-x-4 bottom-6 top-16 flex flex-col justify-end text-white">
            <div className="flex items-center justify-between mb-4">
              <div>
                <p 
                  className="text-sm opacity-90 mb-1 text-brand-body" 
                  style={{ fontFamily: "'Quicksand', sans-serif", fontSize: "0.875rem", fontWeight: 400 }}
                >
                  Your nutrition journey
                </p>
                <h1 
                  className="text-brand-heading" 
                  style={{ fontFamily: "'League Spartan', sans-serif", fontSize: "1.875rem", fontWeight: 700, lineHeight: 1.2 }}
                >
                  Meal Calendar
                </h1>
                <p 
                  className="text-sm opacity-90 mt-1 text-brand-body" 
                  style={{ fontFamily: "'Quicksand', sans-serif", fontSize: "0.875rem", fontWeight: 400 }}
                >
                  Track progress over time
                </p>
              </div>
              <div className="text-center bg-white/20 backdrop-blur-sm rounded-2xl px-4 py-3">
                <div className="flex items-center space-x-2 mb-1">
                  <Calendar className="text-blue-400" size={16} />
                  <span 
                    className="text-2xl font-bold text-brand-heading" 
                    style={{ fontFamily: "'League Spartan', sans-serif", fontSize: "1.5rem", fontWeight: 700 }}
                  >
                    {debugInfo.totalDates}
                  </span>
                </div>
                <p 
                  className="text-xs opacity-90 text-brand-body" 
                  style={{ fontFamily: "'Quicksand', sans-serif", fontSize: "0.75rem", fontWeight: 400 }}
                >
                  logged days
                </p>
              </div>
            </div>
          </div>
        </div>
      </div>

      {/* Debug Panel - Development Only */}
      <div className="bg-yellow-50 border border-yellow-200 rounded-lg p-3 mb-4 text-xs">
        <div className="flex items-center justify-between mb-2">
          <span className="font-semibold text-yellow-800">Debug Info</span>
          <div className="flex gap-2">
            <Button 
              size="sm" 
              variant="outline" 
              onClick={handleRefreshData}
              className="text-xs h-6 px-2"
            >
              <RefreshCw className="w-3 h-3 mr-1" />
              Refresh
            </Button>
            <Button 
              size="sm" 
              variant="outline" 
              onClick={debugLocalStorage}
              className="text-xs h-6 px-2"
            >
              <Database className="w-3 h-3 mr-1" />
              Debug
            </Button>
          </div>
        </div>
        <div className="grid grid-cols-2 gap-2 text-yellow-700">
          <div>dailyMealLogs: {debugInfo.dailyMealLogsCount}</div>
          <div>bytewise-meal-entries: {debugInfo.bytewiseMealEntriesCount}</div>
          <div>Total dates: {debugInfo.totalDates}</div>
          <div>Last load: {debugInfo.lastLoadTime}</div>
        </div>
      </div>

      {/* Calendar Navigation */}
      <div className="flex items-center justify-between mb-4">
        <button
          onClick={() => navigateMonth('prev')}
          className="p-2 hover:bg-muted/50 rounded-lg transition-colors"
        >
          <ChevronLeft size={20} />
        </button>
        <h2 className="text-brand-subheading" style={{ fontFamily: "'Work Sans', sans-serif", fontSize: "1.125rem", fontWeight: 600 }}>
          {monthNames[currentMonth.getMonth()]} {currentMonth.getFullYear()}
        </h2>
        <button
          onClick={() => navigateMonth('next')}
          className="p-2 hover:bg-muted/50 rounded-lg transition-colors"
        >
          <ChevronRight size={20} />
        </button>
      </div>

      {/* Calendar Grid */}
      <div className="bg-card rounded-xl p-4 shadow-sm border border-border mb-6">
        <div className="grid grid-cols-7 gap-1 mb-2">
          {daysOfWeek.map((day) => (
            <div key={day} className="text-center text-xs font-medium text-muted-foreground py-2">
              {day}
            </div>
          ))}
        </div>
        
        <div className="grid grid-cols-7 gap-1">
          {days.map((day, index) => {
            if (!day) {
              return <div key={index} className="aspect-square"></div>;
            }

            const dateKey = formatDateKey(day);
            const dayData = mealData[dateKey];
            const isSelected = isSameDate(day, selectedDate);
            const isTodayDate = isToday(day);

            return (
              <button
                key={index}
                onClick={() => setSelectedDate(day)}
                className={`aspect-square relative rounded-lg transition-all ${
                  isSelected
                    ? 'bg-primary text-primary-foreground'
                    : isTodayDate
                    ? 'bg-primary/20 text-primary font-medium'
                    : 'hover:bg-muted/50'
                }`}
              >
                <span className="text-sm">{day.getDate()}</span>
                {dayData && (
                  <div className={`absolute bottom-0.5 left-1/2 transform -translate-x-1/2 w-1 h-1 rounded-full ${
                    dayData.status === 'complete' ? 'bg-green-500' : 
                    dayData.status === 'partial' ? 'bg-yellow-500' : 'bg-gray-300'
                  } ${isSelected ? 'bg-white' : ''}`}></div>
                )}
              </button>
            );
          })}
        </div>
      </div>

      {/* Selected Date Details - Daily Nutrition Summary */}
      {selectedDateData ? (
        <div className="bg-card rounded-xl p-4 shadow-sm border border-border mb-4">
          <div className="flex items-start gap-3">
            {/* Food image with nutrition-themed visual */}
            <div 
              className="w-16 h-16 flex-shrink-0 overflow-hidden bg-gradient-to-br from-primary/20 to-primary/10 flex items-center justify-center"
              style={{
                clipPath: 'polygon(20% 0%, 80% 0%, 100% 20%, 100% 80%, 80% 100%, 20% 100%, 0% 80%, 0% 20%)'
              }}
            >
              <Utensils size={24} className="text-primary" />
            </div>
            
            <div className="flex-1">
              <div className="flex items-center justify-between mb-2">
                <h3 className="text-brand-subheading" style={{ fontFamily: "'Work Sans', sans-serif", fontSize: "1rem", fontWeight: 600 }}>
                  {selectedDate.toLocaleDateString('en-US', { 
                    weekday: 'long', 
                    month: 'short', 
                    day: 'numeric' 
                  })}
                </h3>
                <Badge 
                  className={`text-xs ${
                    selectedDateData.status === 'complete' 
                      ? 'bg-green-100 text-green-700 border-green-200' 
                      : selectedDateData.status === 'partial'
                      ? 'bg-yellow-100 text-yellow-700 border-yellow-200'
                      : 'bg-gray-100 text-gray-700 border-gray-200'
                  }`}
                  variant="outline"
                >
                  {selectedDateData.status === 'complete' ? 'Complete' : 
                   selectedDateData.status === 'partial' ? 'Partial' : 'No Data'}
                </Badge>
              </div>
              
              {/* Daily Nutrition Summary */}
              <div className="space-y-2 text-sm">
                <div className="grid grid-cols-2 gap-3">
                  <div className="flex items-center justify-between">
                    <span className="text-muted-foreground text-brand-body" style={{ fontFamily: "'Quicksand', sans-serif", fontSize: "0.75rem", fontWeight: 400 }}>
                      Meals logged:
                    </span>
                    <span className="font-medium text-brand-label" style={{ fontFamily: "'Work Sans', sans-serif", fontSize: "0.75rem", fontWeight: 500 }}>
                      {selectedDateData.meals}
                    </span>
                  </div>
                  <div className="flex items-center justify-between">
                    <span className="text-muted-foreground text-brand-body" style={{ fontFamily: "'Quicksand', sans-serif", fontSize: "0.75rem", fontWeight: 400 }}>
                      Calories:
                    </span>
                    <span className="font-medium text-brand-label" style={{ fontFamily: "'Work Sans', sans-serif", fontSize: "0.75rem", fontWeight: 500 }}>
                      {selectedDateData.calories}
                    </span>
                  </div>
                </div>
                
                {/* Macronutrient breakdown */}
                <div className="grid grid-cols-3 gap-2 pt-2 border-t border-border/50">
                  <div className="text-center">
                    <p className="text-xs text-muted-foreground text-brand-body" style={{ fontFamily: "'Quicksand', sans-serif", fontSize: "0.6875rem", fontWeight: 400 }}>
                      Protein
                    </p>
                    <p className="font-medium text-brand-label" style={{ fontFamily: "'Work Sans', sans-serif", fontSize: "0.75rem", fontWeight: 500 }}>
                      {selectedDateData.protein}g
                    </p>
                  </div>
                  <div className="text-center">
                    <p className="text-xs text-muted-foreground text-brand-body" style={{ fontFamily: "'Quicksand', sans-serif", fontSize: "0.6875rem", fontWeight: 400 }}>
                      Carbs
                    </p>
                    <p className="font-medium text-brand-label" style={{ fontFamily: "'Work Sans', sans-serif", fontSize: "0.75rem", fontWeight: 500 }}>
                      {selectedDateData.carbs}g
                    </p>
                  </div>
                  <div className="text-center">
                    <p className="text-xs text-muted-foreground text-brand-body" style={{ fontFamily: "'Quicksand', sans-serif", fontSize: "0.6875rem", fontWeight: 400 }}>
                      Fat
                    </p>
                    <p className="font-medium text-brand-label" style={{ fontFamily: "'Work Sans', sans-serif", fontSize: "0.75rem", fontWeight: 500 }}>
                      {selectedDateData.fat}g
                    </p>
                  </div>
                </div>
                
                {/* Meal breakdown */}
                {selectedDateData.meals > 0 && (
                  <div className="pt-2 border-t border-border/50">
                    <p className="text-xs text-muted-foreground mb-1 text-brand-body" style={{ fontFamily: "'Quicksand', sans-serif", fontSize: "0.6875rem", fontWeight: 400 }}>
                      Meal distribution:
                    </p>
                    <div className="grid grid-cols-4 gap-1 text-xs">
                      {selectedDateData.breakdown.breakfast > 0 && (
                        <div className="text-center">
                          <div className="text-orange-600 font-medium">{selectedDateData.breakdown.breakfast}</div>
                          <div className="text-muted-foreground">Breakfast</div>
                        </div>
                      )}
                      {selectedDateData.breakdown.lunch > 0 && (
                        <div className="text-center">
                          <div className="text-green-600 font-medium">{selectedDateData.breakdown.lunch}</div>
                          <div className="text-muted-foreground">Lunch</div>
                        </div>
                      )}
                      {selectedDateData.breakdown.dinner > 0 && (
                        <div className="text-center">
                          <div className="text-blue-600 font-medium">{selectedDateData.breakdown.dinner}</div>
                          <div className="text-muted-foreground">Dinner</div>
                        </div>
                      )}
                      {selectedDateData.breakdown.snacks > 0 && (
                        <div className="text-center">
                          <div className="text-purple-600 font-medium">{selectedDateData.breakdown.snacks}</div>
                          <div className="text-muted-foreground">Snacks</div>
                        </div>
                      )}
                    </div>
                  </div>
                )}
              </div>
            </div>
          </div>
          
          <Button 
            variant="outline" 
            className="w-full mt-3" 
            size="sm"
            onClick={() => {
              console.log('📊 View Details button clicked - navigating to meal logger');
              if (onNavigate) {
                onNavigate('calculator');
              }
            }}
            style={{ fontFamily: "'Work Sans', sans-serif", fontSize: "0.875rem", fontWeight: 500 }}
          >
            <Target className="mr-2" size={16} />
            View Detailed Nutrition
          </Button>
        </div>
      ) : (
        <div className="bg-card rounded-xl p-6 shadow-sm border border-border mb-4 text-center">
          <div 
            className="w-20 h-20 mx-auto mb-4 bg-muted/30 overflow-hidden flex items-center justify-center"
            style={{
              clipPath: 'circle(40px at center)'
            }}
          >
            <Plus size={32} className="text-muted-foreground" />
          </div>
          <h3 className="text-brand-subheading mb-2" style={{ fontFamily: "'Work Sans', sans-serif", fontSize: "1rem", fontWeight: 600 }}>
            No meals logged
          </h3>
          <p className="text-brand-body text-muted-foreground mb-4" style={{ fontFamily: "'Quicksand', sans-serif", fontSize: "0.875rem", fontWeight: 400 }}>
            Start tracking your nutrition for {selectedDate.toLocaleDateString('en-US', { 
              month: 'short', 
              day: 'numeric' 
            })}
          </p>
          <Button 
            className="bg-primary text-primary-foreground" 
            size="sm"
            onClick={() => {
              console.log('🍽️ Add Meal button clicked - navigating to meal logger');
              if (onNavigate) {
                onNavigate('calculator');
              }
            }}
            style={{ fontFamily: "'Work Sans', sans-serif", fontSize: "0.875rem", fontWeight: 500 }}
          >
            <Plus className="mr-2" size={16} />
            Add Meal
          </Button>
        </div>
      )}

      {/* Weekly Stats with Real Data */}
      <div className="bg-card rounded-xl p-4 shadow-sm border border-border">
        <div className="flex items-center mb-3">
          <TrendingUp className="text-primary mr-2" size={20} />
          <h3 className="text-brand-subheading" style={{ fontFamily: "'Work Sans', sans-serif", fontSize: "1rem", fontWeight: 600 }}>
            This Week
          </h3>
          <Badge variant="outline" className="ml-auto text-xs">
            Live Data
          </Badge>
        </div>
        <div className="grid grid-cols-3 gap-4 text-center">
          <div>
            <div className="text-lg font-semibold text-foreground text-brand-heading" style={{ fontFamily: "'League Spartan', sans-serif", fontSize: "1.125rem", fontWeight: 700 }}>
              {weeklyStats.totalMeals}
            </div>
            <div className="text-xs text-muted-foreground text-brand-body" style={{ fontFamily: "'Quicksand', sans-serif", fontSize: "0.75rem", fontWeight: 400 }}>
              Meals
            </div>
          </div>
          <div>
            <div className="text-lg font-semibold text-foreground text-brand-heading" style={{ fontFamily: "'League Spartan', sans-serif", fontSize: "1.125rem", fontWeight: 700 }}>
              {weeklyStats.avgCalories}
            </div>
            <div className="text-xs text-muted-foreground text-brand-body" style={{ fontFamily: "'Quicksand', sans-serif", fontSize: "0.75rem", fontWeight: 400 }}>
              Avg Calories
            </div>
          </div>
          <div>
            <div className="text-lg font-semibold text-foreground text-brand-heading" style={{ fontFamily: "'League Spartan', sans-serif", fontSize: "1.125rem", fontWeight: 700 }}>
              {weeklyStats.goalMetPercentage}%
            </div>
            <div className="text-xs text-muted-foreground text-brand-body" style={{ fontFamily: "'Quicksand', sans-serif", fontSize: "0.75rem", fontWeight: 400 }}>
              Goal Met
            </div>
          </div>
        </div>
        
        {isLoading && (
          <div className="mt-3 pt-3 border-t border-border/50">
            <div className="flex items-center justify-center space-x-2 text-xs text-muted-foreground">
              <div className="animate-spin rounded-full h-3 w-3 border border-primary border-t-transparent"></div>
              <span>Loading meal data...</span>
            </div>
          </div>
        )}

        {!isLoading && debugInfo.totalDates === 0 && (
          <div className="mt-3 pt-3 border-t border-border/50">
            <div className="flex items-center justify-center space-x-2 text-xs text-yellow-600 bg-yellow-50 rounded p-2">
              <AlertCircle className="w-3 h-3" />
              <span>No meal data found. Try loading demo data or adding meals.</span>
            </div>
          </div>
        )}
      </div>
    </div>
  );
}
